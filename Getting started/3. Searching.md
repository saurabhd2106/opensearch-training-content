# Seaching using Query DSL

- Searching in a particular index

```
GET /blogpost/_search
{
  "query": {
    "match": {
      "description": "search"
    }
  }
}

```


- Searching in all indexes

```
GET /_search
{
  "query": {
    "match": {
      "description": "search"
    }
  }
}

```

```
GET /blogpost/_search
{
  "query": {
    "match_phrase": {
      "description": "Java performance"
    }
  }
}

```

```
GET /blogpost/_search
{
  "query": {
    "match_all": {}
  }
}
```

```
GET /blogpost/_search
{
  "query": {
    "multi_match": {
      "query": "elasticsearch",
      "fields": ["title", "tags"]
    }
  }
}

```

```
GET /blogpost/_search
{
  "query": {
    "bool": {
      "must": [
        {"match": {
          "tags": "Java"
        }}
      ],
      "must_not": [
        {"match": {
          "tags": "elastic"
        }}
      ]
    }
  }
}

```
```
GET /blogpost/_search
{
  "query": {
    "bool": {
      "must": [
        {"match": {
          "tags": "Java"
        }}
      ],
      "must_not": [
```
